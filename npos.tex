  \multicolumn{2}{p{\dimexpr 0.97\linewidth-\tabcolsep-\arrayrulewidth\relax}}{%
    What follows are five subclasses of the class NPO.
    \renewcommand{\labelenumi}{\bf NPO(\Roman{enumi})}
    \begin{enumeratex} \addtolength{\itemindent}{9mm}
    \item Contains every optimization problem from NPO for which there exists a
      FPTAS. \newline Example: Knapsack problem.
    \item Contains every optimization problem from NPO that has a PTAS.
      \newline Example: Makespan scheduling problem.
    \item Contains every optimization problem $U \in NPO$ such that
      \begin{enumeratex}
      \item there is a polynomial-time $\delta$-approximation algorithm for some
        $\delta > 1$, and
      \item there is no polynomial-time $d$-approximation algorithm for $U$ for
        some $d < \delta$ (possibly under some reasonable assumption like $P
        \neq NP$), i.e. there is no PTAS for $U$.
      \end{enumeratex}
      Examples: Min. vertex cover prob, {\sc Max-Sat}, {\sc $\Delta$-TSP}
    \item Contains every $U \in NPO$ such that
      \begin{enumeratex}
      \item there is a polynomial-time $f(n)$-approximation algorithm for $U$
        for some $f : \N \to \R^+$, where $f$ is bounded by a polylogarithmic
        function, and
      \item under some reasonable assumption like $P \neq NP$, there does not
        exist any polynomial-time $\delta$-approximation algorithm for $U$ for
        any $\delta \in \R^+$.
      \end{enumeratex}
      \vspace{-3mm} Examples: Set cover problem
    \item Contains every $U \in NPO$ such that if ther exists a polynomial-time
      $f(n)$-approximation algorithm for $U$, then (under some reasonable
      assumption like $P \neq NP$) $f(n)$ is not bounded by any polylogarithmic
      function.
      \newline Examples: TSP, maximum clique problem
    \end{enumeratex}

  \multicolumn{2}{c}{%
    \bf Approximation Algorithms (cont.)} \\ \hline
  \bf superstable & Let $U = (\Sigma_I, \Sigma_O, L, L_I, \M, cost, goal)$ and
  $\bar{U} = (\Sigma_I, \Sigma_O, L, L, \M, cost, goal)$ be two \\ \cline{1-1}
  \multicolumn{2}{p{\dimexpr 1.32\tabucolX+\tabcolsep+\arrayrulewidth\relax}}{%
    optimization problems with $L_I \subset L$. Let $h$ be a distance function
    for $\bar{U}$ according to $L_I$, and let $U_r = (\Sigma_I, \Sigma_O, L,
    Ball_{r,h}(L_I), \M, cost, goal)$ for every $r \in \R^+$. Let $A =
    \{A_\varepsilon\}_{\varepsilon > 0}$ be a PTAS for $U$.
    If, for every $r > 0$ and every $\varepsilon > 0$, $A_\varepsilon$ is a
    $\delta_{r,\varepsilon}$-approximation algorithm for $U_r$, we say that the
    PTAS $A$ is {\bf stable according to \textit{h}}. \newline
    If $\delta_{r, \varepsilon} \leq f(\varepsilon) \cdot g(r)$, where
    \begin{enumeratex}
    \item $f$ and $g$ are some functions from $\R^{\geq 0}$ to $\R^+$ and
    \item $\lim_{\varepsilon\to0} f(\varepsilon) = 0$
    \end{enumeratex}
    we say that the PTAS $A$ is {\bf superstable according to \textit{h}.}}
  \\ \hline
  \multicolumn{2}{p{\dimexpr 1.32\tabucolX+\tabcolsep+\arrayrulewidth\relax}}{%
    Let $U$ be an optimization problem. A {\bf constraint distance function} for
    $U$ is any function $h : L_I \times \Sigma^*_O \to \R^{\geq 0}$ such that
    \begin{enumeratex}
    \item $h(x, S) = 0$ for every $S \in \M(x)$,
    \item $h(x, S) > 0$ for every $S \notin \M(x)$, and
    \item $h$ is polynomial-time computable
    \end{enumeratex}
    For every $\varepsilon \in \R^+$ and every $x \in L_I$,
    $\mathbf{\M^h_\varepsilon(x)} = \left\{S \in \Sigma_O^* \mid h(x, S) \leq
    \varepsilon\right\}$ is the {\bf $\varepsilon$-ball of $\M(\mathbf{x})$
      according to \textit{h}}} \\ \hline
  \multicolumn{2}{p{\dimexpr 1.32\tabucolX+\tabcolsep+\arrayrulewidth\relax}}{%
    Let $U$ be an optimization problem, nad let $h$ be a constraint distance
    function for $U$. An algorithm $A$ is called {\bf \textit{h}-dual
      polynomial-time approximation scheme (\textit{h}-dual PTAS for
      \textit{U})}, if
    \begin{enumeratex}
    \item for every input $(x, \varepsilon) \in L_I \times \R^+,
      A(x,\varepsilon) \in \M^h_\varepsilon(x)$
    \item $cost(A(x,\varepsilon)) \geq Opt_U(x)$ if $goal = \min$ ($\leq$ for
      $\max$)
    \item $Time_A(x, \varepsilon^{-1})$ is bounded by a fn polynomial in $|x|$
    \end{enumeratex}
    If $Time_A(x, \varepsilon^{-1})$ can be bounded by a fn polynomial in both
    $|x|$ and $\varepsilon^{-1}$, we say that $A$ is a {\bf \textit{h}-dual
      fully polynomial-time approximation scheme (\textit{h}-dual FPTAS) for
      \textit{U}}}
